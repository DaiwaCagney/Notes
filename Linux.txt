DNS:
cat /etc/hosts
cat /etc/resolv.conf
cat /etc/nsswitch.conf

IP address:
ifconfig

Gateway:
netstat -rnv

Process:
ps -ef | grep {process}
ps -ef | more

SSH:
cat /etc/ssh/sshd_conf

Access Log:
cd /var/log
more secure
tail -f secure

ls -ltr

User:
id {username}
vi /etc/passwd

Firewall:
systemctl status iptables.service
systemctl status firewalld
systemctl stop firewalld
systemctl disable firewalld
